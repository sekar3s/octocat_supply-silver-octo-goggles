‚úÖ useMediaQuery Hook
function useMediaQuery(query: string) {
  const [matches, setMatches] = useState(
    () => window.matchMedia(query).matches
  );

  useEffect(() => {
    const media = window.matchMedia(query);
    const listener = () => setMatches(media.matches);
    media.addEventListener('change', listener);
    return () => 
      media.removeEventListener('change', listener);
  }, [query]);

  return matches;
}

4.1.3 Status Messages - No SPA Route Change Announcements

WCAG 4.1.3 ‚Üí
üìç Location
frontend/src/App.tsx
React Router navigation lacks screen reader announcements
üìù Description
When navigating between pages in the SPA, screen readers are not informed that content has changed. Unlike traditional page loads, React Router navigation is silent to assistive technology.
üí• Impact
High Impact: Screen reader users click navigation links and receive no feedback that navigation occurred. They must manually explore the page to discover content changed. This creates significant confusion and inefficiency.
üîß Recommended Fix & Code Examples

Implement a RouteAnnouncer component using ARIA live regions to announce page changes.

‚úÖ RouteAnnouncer Component
// components/RouteAnnouncer.tsx
import { useEffect, useState } from 'react';
import { useLocation } from 'react-router-dom';

const routeTitles: Record<string, string> = {
  '/': 'Home',
  '/products': 'Products',
  '/cart': 'Shopping Cart',
  '/orders': 'Orders',
  '/login': 'Login',
  '/admin': 'Admin Dashboard',
  '/admin/suppliers': 'Manage Suppliers',
};

export function RouteAnnouncer() {
  const location = useLocation();
  const [announcement, setAnnouncement] = 
    useState('');

  useEffect(() => {
    const title = routeTitles[location.pathname] 
      || 'Page';
    
    // Delay to allow page content to render
    const timer = setTimeout(() => {
      setAnnouncement(`Navigated to ${title} page`);
    }, 100);

    return () => clearTimeout(timer);
  }, [location.pathname]);

  return (
    <div
      role="status"
      aria-live="polite"
      aria-atomic="true"
      className="sr-only"
    >
      {announcement}
    </div>
  );
}

}

Focus Visibility - Subtle Default Rings on Light Backgrounds

üìç Location
Global - All interactive elements in light mode
Particularly: form inputs, buttons, links on white/gray-100 backgrounds
üìù Description
While browser default focus indicators technically exist, they are often too subtle on light backgrounds (thin gray outline). Hover states (gray-300) are very close to focus appearance, making it hard to distinguish keyboard focus from mouse hover.
üí• Impact
Medium Impact: Keyboard users struggle to identify focus position, especially on busy pages. Users with low vision may lose track of navigation. Not a WCAG failure if contrast meets 3:1, but significantly impacts usability.
üîß Recommended Enhancement

Implement high-contrast, distinctive focus indicators specifically designed for light mode.

‚úÖ Enhanced Light Mode Focus Styles

  outline-width: 4px;

.light-mode .bg-gray-200 *:focus-visible {
  outline-color: #1E40AF; /* Blue-800 - even darker */
  outline-width: 4px;
}

/* Form inputs get special treatment */
.light-mode input:focus-visible,
.light-mode textarea:focus-visible,
.light-mode select:focus-visible {
  outline: 3px solid #2563EB;
  outline-offset: 0;
  border-color: #2563EB;

  outline: 0; /* Only hide for mouse clicks */

  background-color: #E5E7EB; /* gray-200 */

  /* Focus wins over hover styling */
